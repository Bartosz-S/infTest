import java.awt.event.KeyEvent;
import java.nio.charset.StandardCharsets;

public class CalculatorClass{
    public static int calculate(String str) {
        try {
            return Integer.parseInt(str);
        } catch (NumberFormatException e) {

            //logarytm dziesiÄ™tny
            int indexOf1 = str.indexOf("log(");
            int indexOf2 = str.indexOf(")");
            if(indexOf1 != -1 && indexOf2 != -1){
                String editedStr1 = str.substring(0, indexOf1);
                String editedStr2 = str.substring(indexOf1+4, indexOf2);
                String editedStr3 = str.substring(indexOf2+1);
                int j = calculate(editedStr2);
                int i;
                for(i = 0; j!=1; i++){
                    j/=10;
                }
                String editedStr = editedStr1 + i + editedStr3;
                return calculate(editedStr);
            }

            //nawiasy
            if (str.contains("(") && str.contains(")")) { //ok
                String beforeNawias, nawias, afterNawias;
                beforeNawias = str.substring(0, str.indexOf("("));
                nawias = str.substring(str.indexOf("(")+1, str.indexOf(")"));
                afterNawias = str.substring(str.indexOf(")")+1);
                if (nawias.contains("(")) {
                    beforeNawias = str.substring(0, str.indexOf("("));
                    nawias = str.substring(str.indexOf("(")+1, str.lastIndexOf(")"));
                    afterNawias = str.substring(str.lastIndexOf(")")+1);
                }
                String newNawias = beforeNawias + calculate(nawias) + afterNawias;
                return calculate(newNawias);
            }

            //dodawanie i odejmowanie
            int indexOf = str.indexOf("+");
            if (indexOf != -1) {
                String editedStr1 = str.substring(0, indexOf);
                String editedStr2 = str.substring(indexOf + 1);
                return calculate(editedStr1) + calculate(editedStr2);
            }

            indexOf = str.indexOf("-");
            if(str.contains("-") && str.indexOf("-", indexOf) == indexOf+1){
                if(countMinuses(str)%2 == 0){
                    str = str.substring(countMinuses(str));
                }
                if(countMinuses(str) == 1){
                    str = str.substring(countMinuses(str)-1);
                }
                return calculate(str);
            }

            indexOf = str.indexOf("-");
            if (indexOf != -1) {

                int secondIndexOf = str.indexOf("-", indexOf+1);
                if(secondIndexOf != -1){
                    if(indexOf == 0){
                        String editedStr1 = "(-" + str.substring(indexOf + 1, secondIndexOf) + ")";
                        String editedStr2 = "+(-" + str.substring(secondIndexOf+1) + ")";
                        return calculate(editedStr1+editedStr2);
                    }
                    String editedStr1 = str.substring(0, indexOf);
                    String editedStr2 = "+(-" + str.substring(indexOf + 1, secondIndexOf) + ")";
                    String editedStr3 = "+(-" + str.substring(secondIndexOf+1) + ")";
                    return calculate(editedStr1+editedStr2+editedStr3);
                }
                String editedStr1 = str.substring(0, indexOf);
                String editedStr2 = str.substring(indexOf + 1);
                return calculate(editedStr1) - calculate(editedStr2);
            }

            //mnozenie i dzielenie
            indexOf = str.indexOf("*");
            if (indexOf != -1) {
                String editedStr1 = str.substring(0, indexOf);
                String editedStr2 = str.substring(indexOf + 1);
                return calculate(editedStr1) * calculate(editedStr2);
            }
            indexOf = str.indexOf("/");
            if (indexOf != -1) {
                String editedStr1 = str.substring(0, indexOf);
                String editedStr2 = str.substring(indexOf + 1);
                if((calculate(editedStr1) % calculate(editedStr2) != 0)){
                    System.out.println("reszta:" + calculate(editedStr1)%calculate((editedStr2)));
                }
                return calculate(editedStr1) / calculate(editedStr2);
            }
            indexOf = str.indexOf("^");
            if (indexOf != -1) {
                String editedStr1 = str.substring(0, indexOf);
                String editedStr2 = str.substring(indexOf + 1);
                if(calculate(editedStr2)==0) return 1;
                int l=calculate(editedStr1);
                for (int i = calculate(editedStr2); i > 1; i--) {
                    l*=calculate(editedStr1);
                }
                return l;
            }
            indexOf = str.indexOf("!");
            if (indexOf != -1) {
                String editedStr1 = str.substring(0, indexOf);
                int l=1;
                for(int i = calculate(editedStr1); i > 0; i--){
                    l*=i;
                }
                return l;
            }
            throw new BadFormatExc();
        }
    }

    public static int countMinuses(String string){
        int count = 0;
        int lastIndexOf = -1;
        for(;;){
            if(string.indexOf("-", lastIndexOf) != -1){
                lastIndexOf = string.indexOf("-", lastIndexOf);
                count++;
                continue;
            }
            break;
        }
        return count;
    }
}
