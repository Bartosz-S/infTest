import javax.swing.*;
import java.awt.*;
import java.awt.datatransfer.Clipboard;
import java.awt.datatransfer.DataFlavor;
import java.awt.datatransfer.UnsupportedFlavorException;
import java.awt.event.*;
import java.io.File;
import java.io.FileNotFoundException;
import java.io.IOException;
import java.text.Normalizer;
import java.util.Formatter;


public class App extends JFrame{
    private JPanel panel1;
    private JTextField textField1;
    private JButton buttonSolve;
    private JTextArea textArea1;



    public App() {
        buttonSolve.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                displayResult();
            }
        });
        textField1.addKeyListener(new KeyAdapter() {
            @Override
            public void keyTyped(KeyEvent e) {
                if(!isItProperCharacter(e.getKeyChar())) {
                    e.consume();
                }
            }
        });
        textField1.addKeyListener(new KeyAdapter() {
            @Override
            public void keyPressed(KeyEvent e) {
                if(e.isControlDown() && e.getKeyCode()==KeyEvent.VK_V){
                    Clipboard clipboard = Toolkit.getDefaultToolkit().getSystemClipboard();
                    DataFlavor flavor = DataFlavor.stringFlavor;
                    String input = "";
                    try{
                        input = (String)clipboard.getData(flavor);
                    }
                    catch(UnsupportedFlavorException exception){
                        System.out.println("Not string");
                    }
                    catch(IOException exception){
                        System.out.println("Input/Output Error");
                    }

                    for(int i=0; i<input.length(); i++){
                        if(!isItProperCharacter(input.charAt(i))){
                            e.consume(); break;
                        }
                    }
                }
            }
        });
        textField1.addKeyListener(new KeyAdapter() {
            @Override
            public void keyPressed(KeyEvent e) {
                if(e.getKeyCode()==KeyEvent.VK_ENTER){
                    displayResult();
                }
            }
        });
    }

    boolean isItProperCharacter(char character){
        return (character <= '9' && character >= '0') || character == '+'
                || character == '-' || character == '*' || character == '/'
                || character == '(' || character == ')' || character == '!';
    }

    void displayResult(){
        textArea1.setText("");
        double result = CalculatorClass.calculate(textField1.getText());
        String resultToDisplay = "= "+result;
        textArea1.append(resultToDisplay);
        addLogs();
        generateHistoryLogsOfTheLastUse();
    }

    private String text = "";
    void addLogs(){
        text += (textField1.getText() + textArea1.getText() + "\r\n");
    }

    public void generateHistoryLogsOfTheLastUse(){
        try{
            Formatter f = new Formatter("D:\\obliczenia.txt");
            f.format(text);
            f.close();
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
    }


    public static void main(String args[]){
        JFrame appFrame = new JFrame("Calculator");
        appFrame.getContentPane().add(new App().panel1);
        appFrame.setBounds(0,0,300,200);
        appFrame.setDefaultCloseOperation(3);
        appFrame.setVisible(true);
    }
}
